    roll_no NUMBER PRIMARY KEY,
    name VARCHAR2(20) NOT NULL,
    tamil NUMBER,
    english NUMBER,
    maths NUMBER,
    python NUMBER

COLUMN roll_no FORMAT 999
COLUMN name FORMAT A10
COLUMN tamil FORMAT 999
COLUMN english FORMAT 999
COLUMN maths FORMAT 999
COLUMN python FORMAT 999
COLUMN total FORMAT 999
COLUMN avg FORMAT 999.99
COLUMN percentage FORMAT 999.99
COLUMN grade FORMAT A6
COLUMN result FORMAT A6

SELECT
roll_no,
name,
tamil,
english,
maths,
python,
tamil + english + maths + python AS total,
ROUND((tamil + english + maths + python)/4,2) AS avg,
ROUND(((tamil + english + maths + python)/400)*100,2) AS percentage
FROM marks;
