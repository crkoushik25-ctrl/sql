CREATE TABLE ACTOR (
Act_id INT PRIMARY KEY,
Act_Name VARCHAR(20),
Act_Gender VARCHAR(10)
);
CREATE TABLE DIRECTOR (
Dir_id INT PRIMARY KEY,
Dir_Name VARCHAR(30),
Dir_Phone VARCHAR(15)
);
CREATE TABLE MOVIES (
Mov_id INT PRIMARY KEY,
Mov_Title VARCHAR(30),
Mov_Year INT,
Mov_Lang VARCHAR(15),
Dir_id INT
);
CREATE TABLE MOVIE_CAST (
Act_id INT,
Mov_id INT,
Role VARCHAR(20)
);
CREATE TABLE RATING (
Mov_id INT,
Rev_Stars INT
);

SELECT M.Mov_Title
FROM MOVIES M, DIRECTOR D
WHERE M.Dir_id = D.Dir_id
AND D.Dir_Name = 'Hitchcock';

SELECT DISTINCT M.Mov_Title
FROM MOVIES M, MOVIE_CAST MC
WHERE M.Mov_id = MC.Mov_id
AND MC.Act_id IN
(SELECT Act_id
FROM MOVIE_CAST
GROUP BY Act_id
HAVING COUNT(Mov_id) >= 2);

SELECT DISTINCT A.Act_Name
FROM ACTOR A
JOIN MOVIE_CAST MC1 ON A.Act_id = MC1.Act_id
JOIN MOVIES M1 ON MC1.Mov_id = M1.Mov_id
JOIN MOVIE_CAST MC2 ON A.Act_id = MC2.Act_id
JOIN MOVIES M2 ON MC2.Mov_id = M2.Mov_id
WHERE M1.Mov_Year < 2000
AND M2.Mov_Year > 2015;

SELECT M.Mov_Title,
COUNT(R.Rev_Stars) AS NO_OF_RATINGS,
MAX(R.Rev_Stars) AS HIGHEST_STARS
FROM MOVIES M, RATING R
WHERE M.Mov_id = R.Mov_id
GROUP BY M.Mov_Title
ORDER BY M.Mov_Title;

UPDATE RATING
SET Rev_Stars = 5
WHERE Mov_id IN
(SELECT M.Mov_id
FROM MOVIES M, DIRECTOR D
WHERE M.Dir_id = D.Dir_id
AND D.Dir_Name = 'Steven Spielberg');
SELECT * FROM RATING;
