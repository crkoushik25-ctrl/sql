CREATE TABLE ACTOR (
  Actid NUMBER PRIMARY KEY,
  Actname VARCHAR2(50),
  Actgender VARCHAR2(10)
);

CREATE TABLE DIRECTOR (
  Dirid NUMBER PRIMARY KEY,
  Dirname VARCHAR2(50),
  Dirphone NUMBER
);

CREATE TABLE MOVIES (
  Movid NUMBER PRIMARY KEY,
  Movtitle VARCHAR2(100),
  Movyear NUMBER,
  Movlang VARCHAR2(20),
  Dirid NUMBER,
  FOREIGN KEY (Dirid) REFERENCES DIRECTOR(Dirid)
);

CREATE TABLE RATING (
  Movid NUMBER,
  Revstars NUMBER,
  FOREIGN KEY (Movid) REFERENCES MOVIES(Movid)
);


CREATE TABLE MOVIE_CAST (
  Movid NUMBER,
  Actid NUMBER,
  FOREIGN KEY (Movid) REFERENCES MOVIES(Movid),
  FOREIGN KEY (Actid) REFERENCES ACTOR(Actid)
);
