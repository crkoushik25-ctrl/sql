B.inner joins

CREATE TABLE EMP1 (
EID INT,
ENAME VARCHAR(20),
DID INT
);
CREATE TABLE DEPT (
DID INT,
DNAME VARCHAR(20)
);
INSERT INTO EMP1 VALUES (1,'DHARANESH',10);
INSERT INTO EMP1 VALUES (2,'YOGESH',20);
INSERT INTO EMP1 VALUES (3,'GOKUL',10);
INSERT INTO EMP1 VALUES (4,'VENKY',30);
SELECT * FROM EMP1;

INSERT INTO DEPT VALUES (10,'CSE');
INSERT INTO DEPT VALUES (20,'IT');
INSERT INTO DEPT VALUES (30,'ECE');
SELECT * FROM DEPT;

SELECT ENAME, DNAME
FROM EMP1 E, DEPT D
WHERE E.DID = D.DID;

SELECT E.ENAME, D.DNAME
FROM EMP1 E LEFT JOIN DEPT D
ON E.DID = D.DID;

SELECT E.ENAME, D.DNAME
FROM EMP1 E RIGHT JOIN DEPT D
ON E.DID = D.DID;

SELECT E.ENAME, D.DNAME
FROM EMP1 E FULL OUTER JOIN DEPT D
ON E.DID = D.DID;

groupby and orderby

CREATE TABLE CSD (
EID INT,
DEP VARCHAR(10),
SAL INT
);
INSERT INTO CSD VALUES (1,'CSE',30000);
 INSERT INTO CSD VALUES (2,'IT',25000);
INSERT INTO CSD VALUES (3,'CSE',28000);
INSERT INTO CSD VALUES (4,'ECE',22000);
SELECT * FROM CSD;
SELECT DEP, COUNT(*) FROM CSD GROUP BY DEP;

SELECT DEP, SUM(SAL) FROM CSD GROUP BY DEP;
SELECT DEP, AVG(SAL) FROM CSD GROUP BY DEP;
SELECT DEP, MIN(SAL) FROM CSD GROUP BY DEP;
SELECT DEP, MAX(SAL) FROM CSD GROUP BY DEP;

SELECT * FROM CSD ORDER BY SAL;
SELECT * FROM CSD ORDER BY SAL DESC
