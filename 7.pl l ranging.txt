 create or replace procedure st_report(coursenameinput in varchar2 )
  
    is countfirstclass number :=0;
    countsecondclass number :=0;
    countthirdclass number :=0;
    countfail number :=0;
  
    begin
    select
   count(case when percentage between 70 and 100 then 1 end),
   count(case when percentage between 60 and 69 then 1 end),
  count(case when percentage between 50 and 59 then 1 end),
   count(case when percentage <49 then 1 end)
   into countfirstclass,countsecondclass,countthirdclass,countfail
   from studentcourse
   where coursename=coursenameinput;
 
  dbms_output.put_line('----------STUDENT PERCENTAGE----------');
   dbms_output.put_line('COURSE NAME:'||coursenameinput);
   dbms_output.put_line('100-70%:'||countfirstclass);
   dbms_output.put_line('69-60%:'||countsecondclass);
   dbms_output.put_line('59-50%:'||countthirdclass);
  dbms_output.put_line('BELOW 49%:'||countfail);
   dbms_output.put_line('--------------------------------------');
   end;
   /
